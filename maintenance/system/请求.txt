    SET PHASE_CODE      = 'C',
        STATUS_CODE     = 'D',
        completion_text = '我要亲手灭掉他。'
  WHERE R.REQUEST_ID = 1291122
    AND PHASE_CODE = 'P'
    AND STATUS_CODE IN ('I', 'Q')
    AND (nvl(request_type, 'X') != 'S')
